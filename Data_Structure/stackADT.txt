*연산:

function create(size) do
    최대 크기가 size인 공백 스택을 생성한다.
end

function is_full(s) do
    if(스택의 원소 수==size) do
        return True
    end
    else do
        return False
    end
end

function is_empty(s) do
    if(스택의 원소 수==0) do
        return True
    end
    else do
        return False
    end
end

function push(s,item) do
    if (is_full(s)) do
        return ERROR_STACKFULL
    end
    else do
        스택의 맨 위에 item을 추가한다.
    end
end

function pop(s) do
    if(is_empty(s)) do
        return ERROR_STACKEMPTY
    end
    else do
        스택의 맨 위 원소를 제거해서 return함
    end
end

function peek(s) do
    if(is_empty(s)) do
        return ERROR_STACKEMPTY
    end
    else do
        스택 맨 위 원소를 제거하지 않고 return만 함
    end
end
